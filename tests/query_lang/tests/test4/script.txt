connect to "data_bases";
S -> A | "b" S;
A -> (
    [ "a"+ "b"?
    | "c"
    ] . "a"*
  );
select
    edges
from
    query S intersect with
    graph "data_base_1.txt"
where each edge
    [
        is labeled as "a" ||
        is labeled as S
    ];
select
    edges
from
    query ("c"? "a") named as S intersect with
    graph "data_base_1.txt" where 0..2 are start vertices
where each edge
    [
        is labeled as "a" ||
        isn't finished in start vertices
    ];
select
    count of edges
from
    query S intersect with
    graph "data_base_1.txt" where {1, 3} are final vertices
where each edge
    [
        is labeled as "c" &&
        isn't started in final vertices
    ];
