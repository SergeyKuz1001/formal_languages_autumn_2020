connect to "MiniGDBEvalData";
"Please, choose database name. You can type:
  * G1k-0.001
  * G1k-0.01
  * G1k-0.1
  * G5k-0.001
  * G5k-0.01
  * G10k-0.001
  * G10k-0.01";
ask string Graph_name;
"You've chosen {Graph_name}.txt";

"Where are you?";
ask number ID_Point_From;

"Where do you want to go?";
ask number ID_Point_To;

"Firstly, you must go to this points:";
select final vertices
from
    [
        graph "{Graph_name}.txt" where
            {ID_Point_From} are start vertices
          intersect with
        query "W" as WT
          intersect with
        query "C" as CF
    ]
      intersect with
    [
        graph "{Graph_name}.txt" where
            {ID_Point_To} are final vertices
          intersect with
        query "T" as WT
          intersect with
        query "F" as CF
    ]
where each edge
    (
        is loop and
        (
            is labeled as WT or
            is labeled as CF
        )
    );
